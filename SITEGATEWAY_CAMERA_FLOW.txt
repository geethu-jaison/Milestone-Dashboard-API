5. Registry connection string provider (K/D decrypt)
------------------------------------------------------
You want one place that reads encrypted DB connection from registry values:
- `K` (salted key)
- `D` (encrypted connection string)

Your existing file: `IconnectDashboardGateway.Infrastructure/Connection/RegistryConnectionStringProvider.cs`

Create an interface (recommended) in Application:
`IconnectDashboardGateway.Application/Interfaces/IRegistryConnectionStringProvider.cs` (new)

```csharp

{
    public interface IRegistryConnectionStringProvider
    {
        string GetConnectionString();
    }
}
```

Then implement in Infrastructure (public):
`IconnectDashboardGateway.Infrastructure/Connection/RegistryConnectionStringProvider.cs`

```csharp
using System.Security.Cryptography;
using System.Text;
using Microsoft.Win32;
using IconnectDashboardGateway.Application.Interfaces;

namespace IconnectDashboardGateway.Infrastructure.Connection
{
    public class RegistryConnectionStringProvider : IRegistryConnectionStringProvider
    {
        // Update these to match your plugin registry location:
        private const string RegistryBasePath = @"SOFTWARE\iConnect\Plugin";
        private const string ValueK = "K";
        private const string ValueD = "D";

        private string? _cachedConnectionString;
        private readonly object _lock = new();

        public string GetConnectionString()
        {
            lock (_lock)
            {
                if (!string.IsNullOrWhiteSpace(_cachedConnectionString))
                    return _cachedConnectionString;

                using var key = Registry.LocalMachine.OpenSubKey(RegistryBasePath, writable: false);
                if (key == null)
                    throw new InvalidOperationException($"Registry key not found: {RegistryBasePath}");

                var k = key.GetValue(ValueK) as string;
                var d = key.GetValue(ValueD) as string;
                if (string.IsNullOrWhiteSpace(k) || string.IsNullOrWhiteSpace(d))
                    throw new InvalidOperationException("Registry values K/D are missing or empty.");

                // IMPORTANT:
                // Replace this placeholder with the same decrypt logic used by your plugin scripts.
                // This just shows where K/D decryption would happen.
                var decrypted = DecryptPlaceholder(k, d);

                _cachedConnectionString = decrypted;
                return _cachedConnectionString;
            }
        }

        private static string DecryptPlaceholder(string k, string d)
        {
            // TODO: Use your real algorithm.
            // For now we do a deterministic placeholder to keep the code compiling.
            // Do NOT use in production.
            return $"DECRYPTED_CONNECTION_STRING_FROM_K={k}_D={d}";
        }
    }
}
```

Where to store/cache the connection string:
- keep it in the provider as a singleton-scoped cached value
- do NOT decrypt per request unless required


6. Repository using Dapper and SQL table query
-------------------------------------------------
Update `IconnectDashboardGateway.Infrastructure/DataAccess/CameraRepository.cs`

Example repository:

```csharp
using System.Data;
using System.Data.SqlClient;
using Dapper;
using IconnectDashboardGateway.Application.Interfaces;
using IconnectDashboardGateway.Application.Interfaces.Repositories;
using IconnectDashboardGateway.Contracts.Dtos;

namespace IconnectDashboardGateway.Infrastructure.DataAccess
{
    public class CameraRepository : ICameraRepository
    {
        private readonly IRegistryConnectionStringProvider _connectionStringProvider;

        public CameraRepository(IRegistryConnectionStringProvider connectionStringProvider)
        {
            _connectionStringProvider = connectionStringProvider;
        }

        public async Task<CameraDto> GetCameraSummaryAsync(string siteId, CancellationToken ct)
        {
            var connectionString = _connectionStringProvider.GetConnectionString();

            const string sql = @"
SELECT
    COUNT(1) AS TotalCameras,
    SUM(CASE WHEN IsOnline = 1 THEN 1 ELSE 0 END) AS OnlineCameras
FROM Cameras
WHERE SiteId = @SiteId;";

            await using var conn = new SqlConnection(connectionString);
            await conn.OpenAsync(ct);

            // You will need to adjust column names (TotalCameras/OnlineCameras) to match your DTO mapping.
            return await conn.QuerySingleAsync<CameraDto>(new CommandDefinition(
                sql,
                new { SiteId = siteId },
                cancellationToken: ct
            ));
        }
    }
}
```

Important:
- Replace `Cameras`, `SiteId`, `IsOnline` with your actual table/column names.

using IconnectDashboardGateway.Application.Interfaces;

namespace IconnectDashboardGateway.API.Controllers

    [ApiController]
    [Route("api/v1/sites/{siteId}/cameras")]
    public class CameraController : ControllerBase

            [FromRoute] string siteId,
            CancellationToken ct)

            catch (Exception ex)
            {
                _logger.LogError(ex, "Camera summary failed for siteId={SiteId}", siteId);

                // Optional: you can also call your custom registry+file logger here
                // (see section 8).

                return StatusCode(500, JsonResponseModel<CameraDto>.Error("Internal server error."));
            }
        }

Implement in Infrastructure:
`IconnectDashboardGateway.Infrastructure/Logging/RegistryFileAuditLogger.cs` (new)

```csharp
using System.Text;
using Microsoft.Win32;
using IconnectDashboardGateway.Application.Interfaces;

namespace IconnectDashboardGateway.Infrastructure.Logging
{
    public class RegistryFileAuditLogger : IAppAuditLogger
    {
        // Update these paths/keys to match your environment:
        private const string RegistryPath = @"SOFTWARE\iConnect\SiteGateway\Logs";
        private const string LastErrorValueName = "LastError";
        private const string LastErrorTimeValueName = "LastErrorTime";

        private readonly string _logFilePath;

        public RegistryFileAuditLogger()
        {
            // Example log file. In production, consider making this configurable.
            _logFilePath = Path.Combine(
                AppContext.BaseDirectory,
                "logs",
                "sitegateway.log"
            );
        }

        public void Info(string message)
        {
            AppendLine($"INFO {DateTimeOffset.Now:O} {message}");
        }

        public void Error(string message, Exception? ex = null)
        {
            var full = ex == null ? message : $"{message} | Exception: {ex}";
            AppendLine($"ERROR {DateTimeOffset.Now:O} {full}");

            // Store last error in registry (simple + avoids big registry writes)
            try
            {
                using var key = Registry.LocalMachine.CreateSubKey(RegistryPath);
                if (key != null)
                {
                    key.SetValue(LastErrorValueName, full, RegistryValueKind.String);
                    key.SetValue(LastErrorTimeValueName, DateTimeOffset.Now.ToString("O"), RegistryValueKind.String);
                }
            }
            catch
            {
                // Never fail the API because registry writes failed.
            }
        }

        private void AppendLine(string line)
        {
            Directory.CreateDirectory(Path.GetDirectoryName(_logFilePath)!);
            File.AppendAllText(_logFilePath, line + Environment.NewLine, Encoding.UTF8);
        }
    }
}
```

10. How the request moves through the layers (summary)
---------------------------------------------------------
1. Client calls `GET /api/v1/sites/{siteId}/cameras/summary`
2. `CameraController` receives `siteId`, validates, returns `JsonResponseModel`
3. Controller calls `ICameraService.GetCameraSummaryAsync(siteId, ct)`
4. `CameraService` calls `ICameraRepository.GetCameraSummaryAsync(siteId, ct)`
5. `CameraRepository` asks `IRegistryConnectionStringProvider` for the decrypted connection string (K/D)
6. Repository opens `SqlConnection`, runs Dapper query, maps rows to `CameraDto`
7. Controller wraps DTO into `JsonResponseModel<CameraDto>`
